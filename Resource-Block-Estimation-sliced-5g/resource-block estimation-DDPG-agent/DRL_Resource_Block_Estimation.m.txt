clc; clear; close all;
%% STEP 1: Define RL Environment Parameters
numObservations = 7;  % Includes SliceType
obsInfo = rlNumericSpec([numObservations 1],'LowerLimit',-inf,'UpperLimit',inf);
obsInfo.Name = 'observations';

rb_min = 1;
rb_max = 30;
actInfo = rlNumericSpec([1 1],'LowerLimit',rb_min,'UpperLimit',rb_max);
actInfo.Name = 'RB_Action';

%  Attach environment
env = rlFunctionEnv(obsInfo, actInfo, @mynewStepFunction, @mynewResetFunction);
disp("RL Environment Created (with SliceType).");

%% STEP 2: Actor and Critic Networks
statePath = [
    featureInputLayer(numObservations,'Normalization','none','Name','state')
    fullyConnectedLayer(64,'Name','fc1_state')
    reluLayer('Name','relu1_state')
    fullyConnectedLayer(32,'Name','fc2_state')];

actionPath = [
    featureInputLayer(1,'Normalization','none','Name','action')
    fullyConnectedLayer(32,'Name','fc1_action')];

commonPath = [
    additionLayer(2,'Name','add')
    reluLayer('Name','relu_common')
    fullyConnectedLayer(1,'Name','output')];

criticNetwork = layerGraph();
criticNetwork = addLayers(criticNetwork,statePath);
criticNetwork = addLayers(criticNetwork,actionPath);
criticNetwork = addLayers(criticNetwork,commonPath);
criticNetwork = connectLayers(criticNetwork,'fc2_state','add/in1');
criticNetwork = connectLayers(criticNetwork,'fc1_action','add/in2');

critic = rlQValueRepresentation(criticNetwork,obsInfo,actInfo, ...
    'Observation',{'state'},'Action',{'action'});

actorNetwork = [
    featureInputLayer(numObservations,'Normalization','none','Name','state')
    fullyConnectedLayer(64,'Name','fc1')
    reluLayer('Name','relu1')
    fullyConnectedLayer(32,'Name','fc2')
    reluLayer('Name','relu2')
    fullyConnectedLayer(1,'Name','fc3')
    tanhLayer('Name','tanh_output')
    scalingLayer('Name','scaled_output', ...
        'Scale',(rb_max-rb_min)/2,'Bias',(rb_max+rb_min)/2)];

actor = rlDeterministicActorRepresentation(layerGraph(actorNetwork),obsInfo,actInfo, ...
    'Observation',{'state'},'Action',{'scaled_output'});

disp(" Actor & Critic Networks Created.");

%% STEP 3: Configure Agent
actorOpts = rlOptimizerOptions('LearnRate',1e-4,'GradientThreshold',1);
criticOpts = rlOptimizerOptions('LearnRate',1e-3,'GradientThreshold',1);

agentOpts = rlDDPGAgentOptions(...
    'SampleTime',1, ...
    'TargetSmoothFactor',1e-3, ...
    'ExperienceBufferLength',1e6, ...
    'MiniBatchSize',256, ...
    'DiscountFactor',0.99, ...
    'ActorOptimizerOptions',actorOpts, ...
    'CriticOptimizerOptions',criticOpts);

%  Exploration tweaks

agentOpts.NoiseOptions.Variance = 0.3;        
agentOpts.NoiseOptions.VarianceDecayRate = 5e-6;  
agentOpts.NoiseOptions.MeanAttractionConstant = 0.01;


agent = rlDDPGAgent(actor,critic,agentOpts);
disp(" DDPG Agent Created.");

%% STEP 4: Training (ENABLE RL TRAINING MONITOR)
trainOpts = rlTrainingOptions(...
    'MaxEpisodes',1000, ...
    'MaxStepsPerEpisode',400, ...
    'ScoreAveragingWindowLength',20, ...
    'Verbose',true, ...
    'Plots','training-progress', ...
    'StopTrainingCriteria','none' ...
);

trainingStats = train(agent,env,trainOpts);
save('trainedDDPGAgent_v4.mat','agent');
disp(" Training Completed and Agent Saved.");



%% Manual Learning Curve Plot
figure;
plot(trainingStats.EpisodeIndex, trainingStats.EpisodeReward,'b-','LineWidth',1);
hold on;
plot(trainingStats.EpisodeIndex, trainingStats.AverageReward,'r-','LineWidth',1.5);
xlabel('Episode Number'); ylabel('Reward');
title('DDPG Agent Training Progress');
legend('Episode Reward','Average Reward','Location','southeast');
grid on;
saveas(gcf,'LearningCurve.png');

%% STEP 5: Testing RL Agent vs Rule-Based Allocations
disp(" Loading dataset for testing...");
dataFile = 'ClassifiedDataForDRL.xlsx';
outputFile = 'RL_vs_Rule_Estimation_Output_HYBRID.xlsx';

load('trainedDDPGAgent_v4.mat','agent');
agent.AgentOptions.NoiseOptions.Variance = 0.08;  

data = readtable(dataFile,'VariableNamingRule','modify');
sliceVals = data.SliceType_Output_;
disp(" Dataset loaded for testing.");

%% --- RL Allocations ---
rlRBs = zeros(height(data),1);
staticRuleRBs = zeros(height(data),1);
dynamicRuleRBs = zeros(height(data),1);

% Track RL allocations by slice type
rl_eMBB_RBs = zeros(height(data), 1);
rl_URLLC_RBs = zeros(height(data), 1);
rl_mMTC_RBs = zeros(height(data), 1);

idx_eMBB = 0;
idx_URLLC = 0;
idx_mMTC = 0;

for i = 1:height(data)
    PLR      = toNumeric(data.PacketLossRate_Reliability_(i));
    PDB      = toNumeric(data.PacketDelayBudget_Latency_(i));
    GBR      = toNumeric(data.GBR_Input6_(i));
    UseCase  = toNumeric(data.UseCaseType_Input1_(i));
    Category = toNumeric(data.LTE_5GUECategory_Input2_(i));
    Tech     = toNumeric(data.TechnologySupported_Input3_(i));
    Slice    = encodeSlice(data.SliceType_Output_{i});

    % Normalize state
    state = [PLR; PDB; GBR; UseCase; Category; Tech; Slice];
    state(1) = state(1)/1.0;
    state(2) = state(2)/100.0;
    state(3) = state(3)/100.0;

    %  Get RL Action
    action = getAction(agent,{state});
    rlRBs(i) = action{1};

    % Store RL allocations by slice type
    sliceType = data.SliceType_Output_{i};  
    if strcmpi(sliceType,'eMBB')
    idx_eMBB = idx_eMBB + 1;
    rl_eMBB_RBs(idx_eMBB) = rlRBs(i);

 elseif strcmpi(sliceType,'URLLC')
    idx_URLLC = idx_URLLC + 1;
    rl_URLLC_RBs(idx_URLLC) = rlRBs(i);

 elseif strcmpi(sliceType,'mMTC')
    idx_mMTC = idx_mMTC + 1;
    rl_mMTC_RBs(idx_mMTC) = rlRBs(i);
   end

    if Slice == 3  
        rlRBs(i) = min(rlRBs(i), 10);   
    end

    % Static Rule Allocation
    if strcmpi(sliceType,'eMBB')
        staticRuleRBs(i) = 20;
    elseif strcmpi(sliceType,'URLLC')
        staticRuleRBs(i) = 10;
    elseif strcmpi(sliceType,'mMTC')
        staticRuleRBs(i) = 5;
    end

    % Dynamic Rule Allocation
    if PLR > 0.6 || PDB > 60
        dynamicRuleRBs(i) = 25;
    elseif PLR < 0.2 && PDB < 30
        dynamicRuleRBs(i) = 8;
    else
        dynamicRuleRBs(i) = 15;
    end
end

rl_eMBB_RBs = rl_eMBB_RBs(1:idx_eMBB);
rl_URLLC_RBs = rl_URLLC_RBs(1:idx_URLLC);
rl_mMTC_RBs = rl_mMTC_RBs(1:idx_mMTC);


disp(" Allocations completed.");

%% RL Resource Block Stats per Slice
avg_eMBB_RL  = mean(rl_eMBB_RBs);
avg_URLLC_RL = mean(rl_URLLC_RBs);
avg_mMTC_RL  = mean(rl_mMTC_RBs);

fprintf('\n RL Resource Block Distribution:\n');
fprintf('   eMBB  → Avg RBs = %.2f | Total Samples = %d\n', avg_eMBB_RL, numel(rl_eMBB_RBs));
fprintf('   URLLC → Avg RBs = %.2f | Total Samples = %d\n', avg_URLLC_RL, numel(rl_URLLC_RBs));
fprintf('   mMTC  → Avg RBs = %.2f | Total Samples = %d\n', avg_mMTC_RL, numel(rl_mMTC_RBs));

% RL-only bar chart
figure;
bar([avg_eMBB_RL, avg_URLLC_RL, avg_mMTC_RL]);
set(gca,'XTickLabel',{'eMBB','URLLC','mMTC'});
ylabel('Average RBs Allocated by RL');
title('DDPG Agent RB Distribution Per Slice');
grid on;

%% STEP 6: Save Results
data.RL_RBs = round(rlRBs);
data.Static_Rule_RBs = staticRuleRBs;
data.Dynamic_Rule_RBs = dynamicRuleRBs;
writetable(data, outputFile);
disp(" Results exported to: " + outputFile);

%% STEP 7: Visualization
disp(" Generating visualization plots...");

sliceOrder = {'eMBB','URLLC','mMTC'};

% Averages for RL Agent
% Use the pre-filtered RL allocation arrays for consistency
mean_eMBB_RL   = mean(rl_eMBB_RBs);
mean_URLLC_RL  = mean(rl_URLLC_RBs);

% Averages for Static Rule
mean_eMBB_Static   = mean(staticRuleRBs(strcmpi(sliceVals,'eMBB')));
mean_URLLC_Static  = mean(staticRuleRBs(strcmpi(sliceVals,'URLLC')));
mean_mMTC_Static   = mean(staticRuleRBs(strcmpi(sliceVals,'mMTC')));

% Averages for Dynamic Rule
mean_eMBB_Dyn   = mean(dynamicRuleRBs(strcmpi(sliceVals,'eMBB')));

% Grouped bar data
averageAllocationsPerSlice = [
    mean_eMBB_RL, mean_URLLC_RL, mean_mMTC_RL;
    mean_eMBB_Static, mean_URLLC_Static, mean_mMTC_Static;
    mean_eMBB_Dyn, mean_URLLC_Dyn, mean_mMTC_Dyn
];

% Plot RB Allocation per Slice
figure;
bar(averageAllocationsPerSlice);
set(gca,'XTickLabel',sliceOrder);
ylabel('Average RBs Allocated');
title('Estimated RB Allocation Per Slice (DDPG vs Static vs Dynamic)');
legend('DDPG Agent','Static Rule','Dynamic Rule');

saveas(gcf,'RB_Per_Slice_Comparison.png');

% Overall RB Allocation Comparison

MyNewStepFunction.m

function [nextObs, reward, isDone, loggedSignals] = mynewStepFunction(action, loggedSignals)
    %% Action taken by RL agent
    allocatedRB = action; 
    
    %  Track allocation per slice globally
    global TRACK_eMBB TRACK_URLLC TRACK_mMTC TRACK_step;
    slice = loggedSignals.sliceType;

    if slice == 1
        TRACK_eMBB(end+1) = allocatedRB;
    elseif slice == 2
        TRACK_URLLC(end+1) = allocatedRB;
    elseif slice == 3
        TRACK_mMTC(end+1) = allocatedRB;
    end

    TRACK_step(end+1) = numel(TRACK_step) + 1;
    state = loggedSignals.state;
    noiseLevel = 0.003;
    nextObs = state;
    nextObs(1) = max(0, state(1) - 0.010 * allocatedRB + noiseLevel * randn());   
    nextObs(2) = max(0, state(2) - 0.007 * allocatedRB + noiseLevel * randn());   
    nextObs(3) = min(1, state(3) + 0.0020 * (0.5 - state(3)) * allocatedRB + noiseLevel * randn()); 

    %% Base reward calculation
    qos_reward = 1.2 * (1 - (nextObs(1)^2 + nextObs(2)^2) / 2);  
    gbr_reward = 0.4 * (1 - abs(nextObs(3) - 0.5));           
    rb_penalty = 0.006 * allocatedRB;                            

    %% Slice-specific shaping (Priority: eMBB > URLLC > mMTC)
    slice = loggedSignals.sliceType;

    if slice == 1 
        qos_reward = qos_reward * 2.2;  

        reward = qos_reward + gbr_reward - rb_penalty;

    elseif slice == 2  
        qos_reward = qos_reward * 1.3;   

        % discourage giving URLLC too many RBs
        if allocatedRB > 15
            qos_reward = qos_reward - 0.5 * (allocatedRB - 15);
        end

        reward = qos_reward + gbr_reward - rb_penalty;

    elseif slice == 3   
        rb_penalty = rb_penalty * 6;     

        % Apply SOFT penalty if RBs exceed 10 (NO CAP)
        if allocatedRB > 10
            reward = qos_reward + gbr_reward - rb_penalty - 4 * (allocatedRB - 10);
        else
            reward = qos_reward + gbr_reward - rb_penalty;
        end

        % Bonus for staying efficient (ideal zone: 5–8 RBs)
        if allocatedRB >= 5 && allocatedRB <= 8
            reward = reward + 1.5;
        end
    end

    %% Clamp reward range to avoid instability
    reward = max(min(reward, 6), -6);

    %% Termination condition
    isDone = false;
    if (nextObs(1) > 1.2) || (nextObs(2) > 1.2)
        isDone = true;
        reward = reward - 2;
    end

    %% Update state
    loggedSignals.state = nextObs;

    %% Debugging log (optional)
    persistent stepCounter
    if isempty(stepCounter); stepCounter = 0; end
    stepCounter = stepCounter + 1;
    if mod(stepCounter, 10000) == 0
        fprintf("Step %d | Slice: %d | Reward: %.2f | RBs: %.2f\n", ...
            stepCounter, slice, reward, allocatedRB);
    end
end

myNewResetFunction.m

function [InitialObservation, LoggedSignals] = mynewResetFunction()
    persistent DRLdata rowIndex;

    % Load dataset once
    if isempty(DRLdata)
        disp("Loading and Preparing Data...");
        filePath = 'ClassifiedDataForDRL.xlsx';
        DRLdata = readtable(filePath,'VariableNamingRule','modify');
        rowIndex = 1;
        disp(" Data Loaded.");
    end

    maxTries = 50;
    attempt = 0;

    while attempt < maxTries
        if rowIndex > height(DRLdata)
            rowIndex = 1;
        end

        row = DRLdata(rowIndex,:);
        rowIndex = rowIndex + 1;
        attempt = attempt + 1;

        PLR = toNumeric(row.PacketLossRate_Reliability_);
        PDB = toNumeric(row.PacketDelayBudget_Latency_);

        if PLR <= 1.0 && PDB <= 100
            GBR      = toNumeric(row.GBR_Input6_);
            UseCase  = toNumeric(row.UseCaseType_Input1_);
            Category = toNumeric(row.LTE_5GUECategory_Input2_);
            Tech     = toNumeric(row.TechnologySupported_Input3_);
            Slice    = encodeSlice(row.SliceType_Output_);

            % Normalize values
            PLR_norm = PLR / 1.0;
            PDB_norm = PDB / 100.0;
            GBR_norm = GBR / 100.0;

            % Add very tiny noise to make starting states slightly unique
            PLR_norm = PLR_norm + 0.01*randn();
            PDB_norm = PDB_norm + 0.01*randn();
            GBR_norm = GBR_norm + 0.01*randn();

            InitialObservation = double([PLR_norm; PDB_norm; GBR_norm; UseCase; Category; Tech; Slice]);

            % Log state & slice type for use in step function
            LoggedSignals.state = InitialObservation;
            LoggedSignals.sliceType = Slice;

            %Debug print which slice is initialized (optional)
            fprintf("Reset: Starting new episode with SliceType = %d\n", Slice);
            return;
        end
    end

    % fallback in case dataset is exhausted
    warning(' Fallback to default state.');
    InitialObservation = double([0.2; 0.4; 0.5; 1; 2; 1; 1]);
    LoggedSignals.state = InitialObservation;
    LoggedSignals.sliceType = 1;
end
